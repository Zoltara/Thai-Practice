const getAI = () => {

const getAI = () => {
  const apiKey = import.meta.env.VITE_GEMINI_API_KEY;

  if (!apiKey) {
    throw new Error("VITE_GEMINI_API_KEY is not set");
  }

  return new GoogleGenAI({ apiKey });
};
